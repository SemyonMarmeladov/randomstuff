** Exercise 8
Remove duplicates
#+begin_src haskell
compress :: Eq a => [a] -> [a]
compress xs = foldr f (const []) xs Nothing
  where
    f x r a@(Just q) | x == q = r a
    f x r _ = x : r (Just x)
compress "aaaabbbab"
#+end_src

: Prelude> Prelude> "aaaabbbab"
